% \textbf{Symbols:} $F_t(T_1,T_2)$=fwd rate at $t$ for ops between $T_1,T_2$; $f(t,T)$=inst. fwd rate; $\Phi$=std normal cdf; $i_t(T)$=money mkt rate; $K_f$=fwd swap rate; $\chi^2(x,y,z)$=non-central chi-sq cdf; $P(t,T)$=discount factor/ZC bond; $r_t$=inst. risk-free rate; $\tau$=periodicity/curvature param; $W_t$=BM under physical; $zcy(t,T)$=ZC yield; $Z_t$=BM under EMM.

\textcolor{red}{\textbf{Fundamentals.}}
Discrete compounding: $\prod_{j=0}^{n-1}(1+r_j)$. Continuous: $\exp(\int_0^T r_u du)$. Discrete discount: $\prod_{j=0}^{n-1}(1+r_j)^{-1}$. Continuous: $\exp(-\int_0^T r_u du)$.
Discount factor: $P(t,T)=\mathbb{E}_t^Q[\exp(-\int_t^T r_u du)]$. Properties: $0<P(t,T)<1$, decreasing \& convex in $T$.
ZC yield: $zcy(t,T)=-\frac{1}{T-t}\ln P(t,T)$.
Forward rates (simple): $F(T_1,T_2)=\frac{1}{T_2-T_1}(\frac{1+i(T_2)T_2}{1+i(T_1)T_1}-1)$. Continuous: $F(T_1,T_2)=\frac{1}{T_2-T_1}\ln\frac{P(0,T_1)}{P(0,T_2)}$. From ZCY: $F(T_1,T_2)=\frac{zcy(0,T_2)T_2-zcy(0,T_1)T_1}{T_2-T_1}$.
Inst. fwd rate: $f(T)=\lim_{T_2\to T}F(T,T_2)=-\frac{\partial\ln P(0,T)}{\partial T}$. Relation: $P(0,T)=\exp(-\int_0^T f(u)du)$.
Money mkt instruments: short maturity (<1yr), no interim payments (T-bills, commercial papers, banker's acceptances, munic. notes). Discount basis: $P=M(1-i(T)T)$, yield $y_d=\frac{M-P}{MT}$ [N.Am., UK, EU]. Money-mkt basis: $P=\frac{M}{1+i(T)T}$, yield $y_m=\frac{M-P}{PT}$ [Japan]. Repos: sell securities w/ repurchase commitment.

\textbf{Bonds.} ZC bond: Value $P(0,T)$ for unit face at maturity $T$. Coupon-bearing (par, semi-annual): $\sum_{j=1}^{2T}\frac{c}{2}MP(0,\frac{j}{2})+MP(0,T)$ [$c$=coupon rate, $M$=face, $T$=maturity]. Embedded options: callable (issuer redeems), convertible (holder converts to equity), putable (holder redeems).

\textbf{FRA.} OTC, notional $M$, effective $T_1$, termination $T_2$, ref. rate $i_t$, fixed $R$. Buyer CF at $T_1$: $\frac{M(T_2-T_1)(i_{T_1}(T_2)-R)}{1+i_{T_1}(T_2)(T_2-T_1)}$. If initial value nil: $R=F(T_1,T_2)$. Buyer borrows at $R$ for $[T_1,T_2]$.

\textbf{FRA Futures.} Montreal: CORRA (1mo, 3mo), quoted $100-R$, size $\$2500\times(100-R)$ [\$25/bp]. CME: SOFR (1mo, 3mo), up to 10yr. Hedging: sell $h$ futures (3mo, $M$ mil. loan variable rate): $h=\frac{M\text{ mil.}\times 1\text{ bp}\times 91/365}{25}\approx M$.

\textbf{T-Bill Futures.} CME: quoted $100-R$ (13wk T-bill yield), size $\$2500\times(100-R)$ [\$25/bp], 1yr traded ref. quarters.

\textbf{T-Bond Futures.} Montreal: 4 maturities (2.5,10,20,30yr), \$100k nominal, 6\% coupon, CF adjusts delivered bond to 6\%. Delivery options: timing (any day in delivery month), end-of-month (days between last trading \& delivery), quality (choose bond from pool). Invoice price: $IP_t=H_{t,T}\times CF$. Forward price (cash \& carry): $FP_{t,T}=(SP_t+AC_t)(1+r\frac{T-t}{365})-AC_T-\frac{c}{2}(1+r\frac{T-s}{365})$ [$SP_t$=spot, $AC_t$=accrued int., $s$=last coupon before $T$]. Basis after carry: $BAC_t=FP_{t,T}-IP_t$. CTD bond: minimize $BAC_t$ (maximize $r^*$). Implied repo rate: $r^*=\frac{H_{t,T}\times CF-SP_t+AC_T-AC_t-c/2}{(SP_t+AC_t)\frac{T-t}{365}-\frac{c}{2}\frac{T-s}{365}}$.

\textbf{Options on Bond Futures.} American on bond futures (Montreal: 2,5,10yr), maturities 3mo preceding futures. Weekly also avail.

\textbf{Caps \& Floors.} OTC. Cap=chain of Eur. calls ("caplets"). Floor=chain of Eur. puts ("floorlets"). Contract: notional $M$, ref. $i_t$, strike $K$, maturity $T$, settlements $t_1,\ldots,t_n=T$. Period $\tau=t_j-t_{j-1}$ (typically qtly). Caplet payoff at $t_j$: $M\tau(i_{t_{j-1}}(t_j)-K)^+$. This is put on $P(t_{j-1},t_j)$ w/ strike $K'=\frac{1}{1+K\tau}$, nominal $M'=M(1+K\tau)$, maturity $t_{j-1}$. $\text{Cap}(M,K)=\sum_{j=\alpha+1}^n M'\text{Put}(P(t_{j-1},t_j),K')$. $\text{Floor}(M,K)=\sum_{j=\alpha+1}^n M'\text{Call}(P(t_{j-1},t_j),K')$ [$t_\alpha$=first reset, $t_{\alpha+1}$=first settlement].

\textbf{Swaps.} Collar: Buy cap at $K_2$, sell floor at $K_1$ ($K_1<K_2$). Net at $t_j$: $-M\tau i_{t_{j-1}}(t_j)+M\tau(i_{t_{j-1}}(t_j)-K_2)^+-M\tau(K_1-i_{t_{j-1}}(t_j))^+$. When $K_1=K_2=K$, collar=swap. No-arb: $\text{Cap}(M,i,K)-\text{Floor}(M,i,K)=\text{Payer swap}(M,i,K)$. Payer swap (pays fixed $K$, receives floating): $\text{PS}(M,i,K)=M\tau\sum_{j=\alpha+1}^n P(0,t_j)(F(t_{j-1},t_j)-K)$ where $F(t_{j-1},t_j)=\frac{1}{\tau}(\frac{P(0,t_{j-1})}{P(0,t_j)}-1)$. Simplified: $\text{PS}(M,i,K)=M(P(0,t_\alpha)-P(0,t_n))-M\tau K\sum_{j=\alpha+1}^n P(0,t_j)$. Receiver swap: $\text{RS}(M,i,K)=M(P(0,t_n)-P(0,t_\alpha))+M\tau K\sum_{j=\alpha+1}^n P(0,t_j)$. Fwd swap rate: $K_f=\frac{P(0,t_\alpha)-P(0,t_n)}{\tau\sum_{j=\alpha+1}^n P(0,t_j)}$ (zero value for swaps).

\textbf{Swaptions.} Eur. opt. on swap. Payer swaption: right to enter payer swap. Receiver: right to enter receiver. Maturity=$t_\alpha$, tenor=$t_n-t_\alpha$. Payer payoff: $M\tau\sum_{j=\alpha+1}^n P(t_\alpha,t_j)(F_{t_\alpha}(t_{j-1},t_j)-K)^+$. Swaption=opt. on coupon bond. Swaption value $<$ cap/floor: $(\sum_{j=\alpha+1}^n P(t_\alpha,t_j)(F_{t_\alpha}(t_{j-1},t_j)-K))^+\leq\sum_{j=\alpha+1}^n P(t_\alpha,t_j)(F_{t_\alpha}(t_{j-1},t_j)-K)^+$.

\textbf{Note:} Jan(31), Feb(28), Mar(31), Apr(30), May(31), Jun(30), Jul(31), Aug(31), Sep(30), Oct(31), Nov(30), Dec(31).

\textcolor{red}{\textbf{Yield Curve Smoothing.}}
Money mkt to ZC: For $t$-maturity money mkt rate $R_M(t)$ (Actual/365 Canada, Actual/360 US): $P(0,t)=1-R_M(t)\cdot t$.
Bootstrapping: Long-term swap/bond rates $R_S(T)$ (bullet, semi-annual): $100=\sum_{i=1}^T\frac{100R_S(T)}{2}P(0,\frac{i}{2})+100P(0,T)$. Solve recursively: $P(0,1/2)$, $P(0,1)$ from money mkt. From 2yr rate $R_S(2)$, solve $P(0,2)$. For $P(0,1.5)$, interpolate linearly between $P(0,1)$ and $P(0,2)$.
\textcolor{blue}{\textbf{Nelson-Siegel-Svensson:}} $\text{zcy}_{\text{NSS}}(T)=a+b(\frac{1-e^{-T/\tau}}{T/\tau})+c(\frac{1-e^{-T/\tau}}{T/\tau}-e^{-T/\tau})+d(\frac{1-e^{-T/\theta}}{T/\theta}-e^{-T/\theta})$. Params: $a$=long-end level, $b$=short-end/slope, $c,d$=curvatures, $\tau,\theta$=curvature maturities. Inst. fwd: $f_{\text{NSS}}(T)=a+\frac{e^{-T/\tau}}{T}(b\tau+cT)+\frac{e^{-T/\theta}}{T}dT$. Fitting: $\min_{a,b,c,d,\tau,\theta}\sum_{i=1}^n(\text{zcy}_{\text{NSS}}(t_i)-\text{zcy}(t_i))^2w_i^2$. Weights: $w_i=1$ (none), $w_i=\frac{1/t_i}{\sum_k 1/k}$ (short-end), $w_i=\frac{i}{\sum_k k}$ (long-end).

\textcolor{red}{\textbf{One-Factor Models.}}
\textcolor{blue}{\textbf{Vasicek (equilibrium).}} EMM dynamics: $dr_t=\kappa(\theta-r_t)dt+\sigma dW_t$ [$\kappa$=speed mean reversion, $\theta$=long-run mean, $\sigma$=vol, all $>0$]. Neg. rates possible. $r_T=r_t e^{-\kappa(T-t)}+\theta(1-e^{-\kappa(T-t)})+\sigma\int_t^T e^{-\kappa(T-u)}dW_u$. $\mathbb{E}_t[r_T]=\theta+(r_t-\theta)e^{-\kappa(T-t)}$. $\text{Var}_t[r_T]=\frac{\sigma^2}{2\kappa}(1-e^{-2\kappa(T-t)})$. Physical: $dr_t=\kappa^P(\theta^P-r_t)dt+\sigma dZ_t$ where $\kappa^P=\kappa-\sigma\lambda_1$, $\theta^P=\frac{\kappa\theta+\sigma\lambda_0}{\kappa-\sigma\lambda_1}$. Stationarity: $\lambda_1<\kappa/\sigma$. $P_{vk}(t,T)=a_{vk}(t,T)\exp[-b_{vk}(t,T)r_t]$, $\text{zcy}_k(t,T)=-\frac{1}{T-t}\ln[a_{vk}(t,T))-b_{vk}(t,T)r_t]$ where $a_{vk}(t,T)=\exp[(\theta-\frac{\sigma^2}{2\kappa^2})(b_{vk}(t,T)-T+t)-\frac{\sigma^2}{4\kappa}b_{vk}(t,T)^2]$, $b_{vk}(t,T)=\frac{1-e^{-\kappa(T-t)}}{\kappa}$. Eur. call on \$1 ZC bond (maturity $T$, strike $K$, opt. maturity $T_{\text{call}}$): $\text{call}_{vk}(P,K)=P_{vk}(0,T)\Phi(h)-KP_{vk}(0,T_{\text{call}})\Phi(h-\sigma_p)$, $h=\frac{\sigma_p}{2}+\frac{1}{\sigma_p}\ln\frac{P_{vk}(0,T)}{KP_{vk}(0,T_{\text{call}})}$, $\sigma_p=\frac{\sigma}{\kappa}(1-e^{-\kappa(T-T_{\text{call}})})\sqrt{\frac{1-e^{-2\kappa(T_{\text{call}}-t)}}{2\kappa}}$. $\text{put}_{vk}(P,K)=KP_{vk}(0,T_{\text{put}})\Phi(\sigma_p-h)-P_{vk}(0,T)\Phi(-h)$. Interpretation: Bond value is log-normally distributed. $\sigma_p$=volatility of ZC bond returns; $\Phi(h)$=call delta; $\Phi(h-\sigma_p)$=probability (under EMM) call is exercised. Opt. on coupon bond: $P_c(0,\{T_i\})=\sum_{i=1}^n c_i P(0,T_i)$. Eur. call (strike $K$, maturity $T_{\text{call}}$) payoff: $(\sum_{i=1}^n c_i P(T_{\text{call}},T_i)-K)^+$. Jamshidian decomp: Find $r^*$ where $\sum_{i=1}^n c_i P(T_{\text{call}},T_i,r^*)=K$. Then $\text{call}(P_c(0,\{T_i\}),K)=\sum_{i=1}^n c_i\text{call}(P(0,T_i),K_i)$ where $K_i=P(T_{\text{call}},T_i,r^*)$. Swaptions: Payer swaption=right to enter payer swap (fixed $K$, first reset $t_\alpha$, settlements $t_{\alpha+1},\ldots,t_n$). Pseudo-coupon bond: $c_i=K\tau$ for $i=\alpha+1,\ldots,n-1$; $c_n=1+K\tau$. Find $r^*$ (ATM): $\sum_{i=\alpha+1}^n c_i P(t_\alpha,t_i,r^*)=1$. $\text{payer swaption}(M,K,t_\alpha)=M\sum_{i=\alpha+1}^n c_i\text{put}(P(t_\alpha,t_i),K_i)$. $\text{receiver swaption}(M,K,t_\alpha)=M\sum_{i=\alpha+1}^n c_i\text{call}(P(t_\alpha,t_i),K_i)$ where $K_i=P(t_\alpha,t_i,r^*)$.

\textcolor{blue}{\textbf{CIR (equilibrium).}} EMM: $dr_t=\kappa(\theta-r_t)dt+\sigma\sqrt{r_t}dW_t$. Params as Vasicek. Square-root ensures $r_t\geq 0$ if $r_0\geq 0$ \& Feller: $2\kappa\theta>\sigma^2$. Distribution: $r_t=Y/b_t$ where $b_t=\frac{4\kappa}{\sigma^2(1-e^{-\kappa t})}$, $Y\sim\chi^2_\nu(\lambda_c)$ w/ $\nu=4\kappa\theta/\sigma^2$ d.o.f., non-centrality $b_t r_0 e^{-\kappa t}$. Physical: $dr_t=\kappa^P(\theta^P-r_t)dt+\sigma\sqrt{r_t}dZ_t$ where $\kappa^P=\kappa-\sigma\lambda$, $\theta^P=\frac{\kappa\theta}{\kappa-\sigma\lambda}$. Stationarity: $\lambda<\kappa/\sigma$. ZC bond: $P_{\text{cir}}(t,T)=a_{\text{cir}}(t,T)\exp[-b_{\text{cir}}(t,T)r_t]$, $a_{\text{cir}}(t,T)=(\frac{2\gamma e^{\frac{\kappa+\gamma}{2}(T-t)}}{(\kappa+\gamma)(e^{\gamma(T-t)}-1)+2\gamma})^{\frac{2\kappa\theta}{\sigma^2}}$, $b_{\text{cir}}(t,T)=\frac{2(e^{\gamma(T-t)}-1)}{(\kappa+\gamma)(e^{\gamma(T-t)}-1)+2\gamma}$, $\gamma=\sqrt{\kappa^2+2\sigma^2}$. $\text{zcy}(t,T)=-\frac{1}{T-t}[\ln(a_{\text{cir}}(t,T))-b_{\text{cir}}(t,T)r_t]$. Eur. call (strike $K$, maturity $T_{\text{call}}$): $\text{call}_{\text{cir}}(P,K)=P_{\text{cir}}(0,T)\chi^2(\frac{2\xi\ln(a_{\text{cir}}(T_{\text{call}},T)/K)}{b_{\text{cir}}(T_{\text{call}},T)},\frac{4\kappa\theta}{\sigma^2},2\phi^2re^{\gamma T_{\text{call}}})-KP_{\text{cir}}(0,T_{\text{call}})\chi^2(\frac{2\ln(a_{\text{cir}}(T_{\text{call}},T)/K)}{b_{\text{cir}}(T_{\text{call}},T)(\phi+\psi)},\frac{4\kappa\theta}{\sigma^2},\frac{2\phi^2re^{\gamma T_{\text{call}}}}{\phi+\psi})$ where $\phi=\frac{2\gamma}{\sigma^2(e^{\gamma T_{\text{call}}}-1)}$, $\psi=\frac{\gamma+\kappa}{\sigma^2}$, $\xi=\phi+\psi+b_{\text{cir}}(T_{\text{call}},T)$.

\textcolor{blue}{\textbf{Hull-White (arbitrage).}} $dr_t=\kappa(\varphi(t)-r_t)dt+\sigma dW_t$. $\varphi(t)$ (deterministic) calibrates to observed curve at $t=0$. At $t=0$: $P_{\text{hw}}(0,T)=P^m(0,T)$ (observed). ZC bond: $P_{\text{hw}}(t,T)=a_{\text{hw}}(t,T)\exp[-b_{\text{hw}}(t,T)r_t]$ where $b_{\text{hw}}(t,T)=\frac{1-e^{-\kappa(T-t)}}{\kappa}$ (same as Vasicek), $a_{\text{hw}}(t,T)=\frac{P^m(0,T)}{P^m(0,t)}\exp[b_{\text{hw}}(t,T)f^m(0,t)-\frac{\sigma^2}{4\kappa}b_{\text{hw}}(t,T)^2(1-e^{-2\kappa t})]$. Other formulas follow Vasicek.

\textcolor{blue}{\textbf{Brigo-Mercurio (arbitrage).}} Shifted CIR: $dx_t=\kappa(\theta-x_t)dt+\sigma\sqrt{x_t}dW_t$, $r_t=x_t+\varphi(t)$. Fits curve when $\varphi(t)=f^m(0,t)-f^{cir}(0,t)$. WARNING: doesn't guarantee positive rates. Positivity if $\varphi(t)>0$ $\forall t$. ZC bond: $P_{\text{bm}}(t,T)=a_{\text{bm}}(t,T)\exp[-b_{\text{bm}}(t,T)x_t]$ where $b_{\text{bm}}(t,T)=b_{\text{cir}}(t,T)$, $a_{\text{bm}}(t,T)=\frac{P^m(0,T)}{P^m(0,t)}\frac{a_{\text{cir}}(0,t)e^{-b_{\text{cir}}(0,t)x_0}}{a_{\text{cir}}(0,T)e^{-b_{\text{cir}}(0,T)x_0}}a_{\text{cir}}(t,T)e^{\varphi(t)(T-t)}$. Options follow CIR.

\textcolor{blue}{\textbf{Black-Karasinski (arbitrage).}} Log-normal: $d\ln(r_t)=\kappa(\varphi(t)-\ln(r_t))dt+\sigma dW_t$. $r_t$ is log-normal \& always positive. Let $x_t=\ln(r_t)$, then $x_t$ follows Vasicek w/ mean-reverting level $\varphi(t)$. Thus $r_t=\exp(x_t)$. By Itô: $\frac{dr_t}{r_t}=(\kappa\varphi(t)-\kappa\ln(r_t)+\frac{\sigma^2}{2})dt+\sigma dW_t$. Solution: $r_t=\exp[\ln(r_s)e^{-\kappa(t-s)}+\kappa\int_s^t\varphi(u)e^{-\kappa(t-u)}du+\sigma\int_s^t e^{-\kappa(t-u)}dW_u]$.

\textcolor{red}{\textbf{Multifactor Models.}}
\textcolor{blue}{\textbf{Two-Factor Gaussian (Hull-White 1994/Brigo-Mercurio 2006).}} Factors: $dx_t=-ax_t dt+\sigma dW_{1t}$, $dy_t=-by_t dt+\eta dW_{2t}$ ($x_0=y_0=0$, $W_{1t},W_{2t}$ corr. $\rho$). Short rate: $r_t=x_t+y_t+\varphi(t)$. ZC bond: $P_{2fg}(t,T)=\frac{P^m(0,T)}{P^m(0,t)}\exp(B(t,T))$, $B(t,T)=\frac{1}{2}[V(t,T)-V(0,T)+V(0,t)]-\frac{1-e^{-a(T-t)}}{a}x_t-\frac{1-e^{-b(T-t)}}{b}y_t$. Eur. call: $\text{call}_{2fg}(P,K)=P_{2fg}(0,T)\Phi(h)-KP_{2fg}(0,T_{\text{call}})\Phi(h-\sigma_p)$ where $h=\frac{\sigma_p}{2}+\frac{1}{\sigma_p}\ln\frac{P_{2fg}(0,T)}{KP_{2fg}(0,T_{\text{call}})}$, $\sigma_p$ from $V(t,T)$. Two-factor captures level \& slope better than one-factor.

\textcolor{blue}{\textbf{Fong-Vasicek.}} Two-factor Vasicek w/ rate vol. as 2nd factor. EMM: $dr_t=\kappa(\theta-r_t)dt+\sqrt{v_t}dZ_{1t}$, $dv_t=\gamma(\bar{v}-v_t)dt+\xi\sqrt{v_t}dZ_{2t}$ ($Z_{1t},Z_{2t}$ corr. $\rho$). $r_t$=short rate, $v_t$=variance. ZC bond: $P_{fvk}(t,T)=\exp[-b_{fvk,1}(t,T)r_t+b_{fvk,2}(t,T)v_t+b_{fvk,3}(t,T)]$ w/ $b_{fvk,1}(t,T)=b_{vk}(t,T)=\frac{1-e^{-\kappa(T-t)}}{\kappa}$, $b_{fvk,2}(t,T)$, $b_{fvk,3}(t,T)$ via confluent hypergeometric funcs (\textcolor{blue}{Selby-Strickland 1995} recursive alg.). Stochastic vol improves fit vs constant-vol Vasicek.

\textcolor{blue}{\textbf{Longstaff-Schwartz.}} CIR w/ stochastic vol. EMM: $dr_t=\mu_r(r_t,v_t)dt+a\sqrt{\frac{\beta r_t-v_t}{a(\beta-a)}}dZ_{1t}+\beta\sqrt{\frac{v_t-ar_t}{\beta(\beta-a)}}dZ_{2t}$, $dv_t=\mu_v(r_t,v_t)dt+a^2\sqrt{\frac{\beta r_t-v_t}{a(\beta-a)}}dZ_{1t}+\beta^2\sqrt{\frac{v_t-ar_t}{\beta(\beta-a)}}dZ_{2t}$ ($Z_{1t},Z_{2t}$ indep.). Vol constraint: $a\tau\leq v_t\leq b\tau$. $\beta$ large relative to $\alpha$ for stochastic vol. ZC bond: $P_{ls}(t,T)=a_{ls,1}(t,T)^y a_{ls,2}(t,T)^z\exp[b_{ls,1}(t,T)r_t+b_{ls,2}(t,T)v_t+b_{ls,3}(t,T)]$. Eur. call: $\text{call}_{ls}(P,K)=P_{ls}(0,T)\chi^2(\cdot;\cdot)-KP_{ls}(0,T_{\text{call}})\chi^2(\cdot;\cdot)$.

\textcolor{blue}{\textbf{Two-Factor CIR (Brigo-Mercurio).}} $dx_t=\kappa_1(\theta_1-x_t)dt+\sigma_1\sqrt{x_t}dZ_{1t}$, $dy_t=\kappa_2(\theta_2-y_t)dt+\sigma_2\sqrt{y_t}dZ_{2t}$ ($Z_{1t},Z_{2t}$ indep.). Feller: $2\kappa_i\theta_i\geq\sigma_i^2$ ($i=1,2$) ensures $x_t,y_t\geq 0$. Short rate: $r_t=x_t+y_t$. Arb-free: $r_t=x_t+y_t+\varphi(t)$ w/ $\varphi(t)=f^m(0,t)-f^{cir}_1(0,t)-f^{cir}_2(0,t)$. ZC bond: $P_{2fcir}(t,T)=a_{cir,1}(t,T)a_{cir,2}(t,T)\exp[-b_{cir,1}(t,T)x_t-b_{cir,2}(t,T)y_t]$ w/ $a_{cir,i}(t,T)$, $b_{cir,i}(t,T)$ from one-factor CIR, params $\{\kappa_i,\theta_i,\sigma_i\}$ for $i=1,2$.

\textcolor{blue}{\textbf{N-Factor.}} Each factor: $dx_{it}=\kappa_i(\theta_i-x_{it})dt+\sigma_i\sqrt{x_{it}}dZ_{it}$ (indep. BM). Short rate: $r_t=\sum_{i=1}^N x_{it}$ (or $+\varphi(t)$ arb-free). ZC bond: product of $N$ one-factor CIR bonds.

\textcolor{blue}{\textbf{Affine Term Structure (Duffie-Kan).}} $N$ factors $X_t=(X_{1t},\ldots,X_{Nt})'\in\mathbb{R}^N$. EMM: $dX_t=(K\theta-KX_t)dt+S_t dW_t$ where $K$ is $N\times N$, $\theta$ is $N\times 1$, $W_t$ $N$-dim BM, $S_t$ $N\times N$ diagonal w/ $[S_t]_{ii}=\sqrt{a_i+\beta_i'X_t}$ [$a_i\geq 0$, $\beta_i\in\mathbb{R}^N$]. Market price of risk: $\lambda_t=\lambda_1 S_t+S_t\lambda_2 X_t$ [$\lambda_1\in\mathbb{R}^N$, $\lambda_2$ $N\times N$]. Linear short rate: $r_t=\delta_0+\delta_1'X_t$ [$\delta_0\in\mathbb{R}$, $\delta_1\in\mathbb{R}^N$]. ZC bond (exponential-affine): $P_{aff}(t,T)=\exp[A(\tau)-B(\tau)'X_t]$ [$\tau=T-t$].

\textcolor{red}{\textbf{Implementation.}}
Estimation: fit to historical curve (ex-post). Calibration: fit to contemporaneous data (yields+derivatives) for precise pricing. Procedure: equilibrium models fit to yield curve (min sq. errors); arbitrage models fit to yield then caps/floors or swaption prices. Check price fit, more stringent: check implied vols (using Black's) fit.

\textcolor{blue}{\textbf{Black for Caps/Floors.}} Assumes fwd rates lognormal; ignores corr. between discount \& payoff; future spot rate $\approx$ future fwd rate $l_{t_{j-1}}(t_j)=F(t_{j-1},t_j)$. Cap (strike $K$, maturity $t_n$): $\text{cap}(t_n)=M\sum_{j=i+1}^n\tau P(0,t_j)[F(t_{j-1},t_j)\Phi(d_1)-K\Phi(d_2)]$ [$\tau$=accrual (typ. 0.25 qtly), $M$=notional]. $d_1=\frac{1}{v_j}[\ln\frac{F(t_{j-1},t_j)}{K}+\frac{v_j^2}{2}]$, $d_2=d_1-v_j$, $v_j=\sigma_{\text{cap}}\sqrt{t_{j-1}}$ [$\sigma_{\text{cap}}$=implied vol from mkt, varies by tenor; first reset $t_0=0.25$, $t_1=0.5$, etc.]. Floor: $\text{floor}(t_n)=M\sum_{j=i+1}^n\tau P(0,t_j)[K\Phi(-d_2)-F(t_{j-1},t_j)\Phi(-d_1)]$. ATM strike: $K=K_f$ (fwd swap rate) when cap/floor ATM: $K_f=\frac{P(0,t_a)-P(0,t_n)}{\sum_{j=i+1}^n\tau P(0,t_j)}$.

\textcolor{blue}{\textbf{Black for Swaptions.}} Ignores corr. between discount \& payoff; assumes fwd swap rates driftless GBM under EMM. Payer swaption (strike $K$, maturity $t_n$): $\text{payer swaption}(t_n)=M(K_f\Phi(d_1)-K\Phi(d_2))\sum_{j=a+1}^n\tau P(0,t_j)$. Receiver: $\text{receiver swaption}(t_n)=M(K\Phi(-d_2)-K_f\Phi(-d_1))\sum_{j=a+1}^n\tau P(0,t_j)$ where $K_f$=fwd swap rate. $d_1=\frac{1}{v}[\ln\frac{K_f}{K}+\frac{v^2}{2}]$, $d_2=\frac{1}{v}[\ln\frac{K_f}{K}-\frac{v^2}{2}]$, $v=\sigma_{\text{swapt}}\sqrt{t_a}$ [$\sigma_{\text{swapt}}$=implied swaption vol from mkt, $t_a$=swap start].

\textcolor{blue}{\textbf{Kalman Filter.}} State-space: measurement eq. relates observed ZCY to hidden states; transition eq. describes state evolution; estimates params maximizing log-likelihood. Measurement: $ZCY_t=A+BX_t+\eta_t$ [$A$ $N\times 1$, $B$ $N\times K$, $\eta_t$ $N\times 1$ errors w/ diag. cov. $H=\text{diag}(h_{t_1}^2,\ldots,h_{t_N}^2)$; variances may differ by maturity but constant over time; uncorrelated cross-sectionally]. Transition: $x_{i,t}=\mathbb{E}[x_{i,t}|x_{i,t-\delta}]+\sqrt{\text{Var}[x_{i,t}|x_{i,t-\delta}]}\xi_t$. Algorithm: loop over obs. dates: (1) predict next-period state estimators \& error variance; (2) compute prediction error (observed-predicted yields) \& its variance; (3) compute log-likelihood from prediction error; (4) update state estimators using prediction error \& updated variance. Final params maximize total log-likelihood.

\textcolor{blue}{\textbf{Diebold-Li.}} Dynamic Nelson-Siegel. Time-varying loadings $L_a=1$, $L_b=\frac{1-e^{-\tau/\tau}}{T}$, $L_c=(\frac{1-e^{-\tau/\tau}}{T}-e^{-\tau/T})$ [param $\tau$ fixed]. Factor representation: $zcy_t(T)=a_t L_a+b_t L_b+c_t L_c$ [$a_t$=level, $b_t$=slope, $c_t$=curvature]. Forecasting: Set $\tau$ so curvature loading $L_c$ max at 30mo. AR(1) for each factor: $\hat{x}_{t+h}=c_x+k_x\hat{x}_t$ for $x=\{a,b,c\}$. Param estimation: OLS regression of $x_{t+h}$ on $x_t$ estimates intercept $c_x$ \& slope $k_x$ minimizing sq. residuals between observed future factor values \& predicted values.

\textcolor{red}{\textbf{HJM \& Market Models.}}
\textcolor{blue}{\textbf{HJM Framework (Heath-Jarrow-Morton 1992).}} Models entire yield curve via inst. fwd rates $f(t,T)$ rather than short rate $r_t$. Key: specifying fwd rate vol. function $\sigma(t,T)$ fully determines dynamics of whole yield curve. HJM drift condition: Under EMM, drift entirely pinned down by vol—no free drift params. Fwd rate dynamics: $df(t,T)=\sigma(t,T)\int_t^T\sigma(t,u)du\,dt+\sigma(t,T)dW_t$. Specifying $\sigma(t,T)$ characterizes model for all maturities simultaneously. Markov condition (\textcolor{blue}{Carverhill 1994}): $r_t=f(t,t)$ is Markovian iff $\sigma(t,T)=h(t)\cdot g(T)$ (separable). Then HJM reduces to generalized Hull-White, implementable on recombining tree. Vol specs: Ritchken-Sankarasubramanian: $\sigma(t,T)=\psi_t\exp(-\kappa(T-t))$ ($\psi_t$ stochastic), leads to 2-state Markov $(r_t,\phi_t)$ for tree/PDE. Li-Ritchken-Sankarasubramanian: $\sigma(t,T)=v_t^\beta\exp(-\kappa(T-t))$, $0\leq\beta\leq 1$, interpolates Gaussian ($\beta=0$) to log-normal ($\beta=1$). Mercurio-Moraleda/Moraleda-Vorst: polynomial-exponential forms generate humped term structure of vol (vol peaks at intermediate maturities)—stylized fact of cap implied vol surfaces. Former lacks closed-form ZCB; latter has closed-form ZCB.

\textcolor{blue}{\textbf{Market Models (BGM/LMM).}} Motivation: Mkt quotes cap/floor/swaption as Black implied vols, but Black rests on invalid assumptions (ignores discount-payoff corr.; assumes driftless GBM for fwd rates under EMM). Market models rigorously justify Black via change of prob. measure. Key: Simply-compounded fwd rate $F(t;T_1,T_2)$ is ratio of traded assets $/P(t,T_2)$, so it's $T_2$-fwd martingale under $T_2$-fwd measure $\mathbb{Q}^{T_2}$. Under this measure can model as driftless GBM, exactly recovering Black caplet formula w/o approx.

\textcolor{blue}{\textbf{Cap Mkt Model (BGM/Miltersen-Sandmann-Sondermann/Jamshidian 1997).}} Each fwd rate $F(t;T_{k-1},T_k)$ follows driftless GBM under own fwd measure $\mathbb{Q}^{T_k}$. Caplets priced exactly w/ Black; cap=sum Black caplets. Calibration: fit vol. function $\gamma(t,T_k)$ (e.g., parametric humped $\gamma=[\alpha+\beta(T_k-t)]e^{-\delta(T_k-t)}+\theta$) to mkt cap implied vols minimizing sq. pricing errors. Swaptions can't be priced analytically in cap mkt model $\Rightarrow$ use MC simulation. Under diff. fwd measure $\mathbb{Q}^{T_n}$, fwd rates acquire drift correction; simulation uses drift-adjusted dynamics via Euler scheme.

\textcolor{blue}{\textbf{Swap Mkt Model (Jamshidian 1997).}} Fwd swap rate $K_f(t)$ is ratio of traded assets, hence martingale under swap measure (annuity as numéraire). Models $K_f(t)$ as driftless GBM $\Rightarrow$ Black swaption formula exact. Cap/floor mkt model \& swap mkt model theoretically incompatible: can't simultaneously use Black for both caps \& swaptions in single consistent model. Practitioners choose one \& price other numerically.

\textcolor{blue}{\textbf{Backward-Looking Rates (Post-LIBOR, Lyashenko-Mercurio 2019).}} W/ LIBOR end, caps/floors/swaptions reference compounded overnight rates set in arrears (SOFR, CORRA). Backward-looking: rate for $[T_1,T_2]$ not known until $T_2$. Key: Extend ZC bond def. beyond maturity: for $t>T$, set $P(t,T)=B(t)/B(T)$ (bank account ratio), remains valid numéraire. Backward-looking fwd rate $\tilde{F}(t;T_1,T_2)$ still $T_2$-martingale, giving formula analogous to fwd-looking case. Vol adjustment: Before $T_1$, $\tilde{F}(t;T_1,T_2)$ behaves like fwd-looking rate (same vol $\gamma(t,T_2)$). Between $T_1$ \& $T_2$, compounding overnight rates induces linear vol decay: $\bar{\gamma}(t;T_1,T_2)=\min(1,\frac{T_2-t}{T_2-T_1})\gamma(t,T_2)$. Market model structure \& measure-change drift corrections carry over unchanged.

\textcolor{red}{\textbf{Numerical Methods.}}
\textcolor{blue}{\textbf{Binomial Trees (Heston 1995).}} Discrete-time arb-free tree for one-factor where short rate follows random walk w/ mean reversion. Dynamics: $r_{t+\Delta t}=\alpha+\rho r_t+\sigma\varepsilon$. Connection to continuous: equiv. to Vasicek/CIR w/ $\kappa=1-\rho$, $\theta=\alpha/\kappa$. No-arb valuation: $g(t)=e^{-r_t\Delta t}[\pi g_u(t+\Delta t)+(1-\pi)g_d(t+\Delta t)]$ w/ risk-neutral prob. $\pi=\frac{e^{r_t\Delta t}-d}{u-d}$, $u=\exp[(-\alpha-\rho r_t-\sigma)\Delta t]$, $d=\exp[(-\alpha-\rho r_t+\sigma)\Delta t]$. Ensures martingale: discounted value process is martingale under EMM.

\textcolor{blue}{\textbf{Trinomial Trees (Hull-White).}} Richer dynamics, faster convergence; more complex but better for mean-reverting. Designed for Hull-White (adaptable for log-normal like Black-Karasinski). Underlying: $dx_t=-\kappa x_t dt+\sigma dW_t$. Normally distributed, mean-reverting to zero. Tree structure: Each time step $\Delta t$, discretized value $x_{i,j}$ ($i$=time, $j$=space) branches to 3 nodes w/ transition probs $\pi_u,\pi_m,\pi_d$: Up: $x_{i+1,j+1}$; Middle: $x_{i+1,j}$; Down: $x_{i+1,j-1}$. Prob. constraints \& moment matching: $\pi_u+\pi_m+\pi_d=1$, $\pi_u(x_{i,j}+\Delta x_j)+\pi_m x_{i,j}+\pi_d(x_{i,j}-\Delta x_j)=x_{i,j}e^{-\kappa\Delta t}$. Variance: $\pi_u(x_{i,j}+\Delta x_j)^2+\pi_m x_{i,j}^2+\pi_d(x_{i,j}-\Delta x_j)^2=[x_{i,j}+\Delta x_j(\pi_u-\pi_d)]^2+\sigma^2\frac{1-e^{-2\kappa\Delta t}}{2\kappa}$. Branching geometry: adapts in regions to improve mean reversion \& reduce neg. rates. Process (A): normal branching, states $\{x_{i,j}-\Delta x,x_{i,j},x_{i,j}+\Delta x\}$. Process (B): $x_{i,j}$ too high (pulls down), states $\{x_{i,j}-2\Delta x,x_{i,j}-\Delta x,x_{i,j}\}$. Process (C): $x_{i,j}$ too low (pulls up), states $\{x_{i,j},x_{i,j}+\Delta x,x_{i,j}+2\Delta x\}$. Critical levels $j_{\min}$, $j_{\max}$ determine branching switches.

\textcolor{blue}{\textbf{2nd Stage: Calibration via Arrow-Debreu.}} Calibrate tree nodes to match mkt ZC bond prices using Arrow-Debreu prices $Q_{ij}$ (state prices: value of claim paying \$1 at node $j$, time $i$). ZC bond pricing constraint w/ shift param: $P_{NSS}(0,(i+1)\Delta t)=\sum_k Q_{ik}\exp(-\alpha_i\Delta t)\pi_k$. Solve for shift param $\alpha_i$ to match observed bond price $P_{NSS}(0,(i+1)\Delta t)$ from mkt. State price propagation: $Q_{(i+1)j}=\sum_k Q_{ik}\pi_{kj}\exp(-(\alpha_i+k\Delta x)\Delta t)$ where $\pi_{kj}$=transition prob. from state $k$ to $j$. Black-Karasinski extension: For log-normal rates, apply $f^{-1}(\cdot)=\exp(\cdot)$ \& solve $\alpha_i$ numerically (nonlinear). Key: Match each bond maturity sequentially; one $\alpha_i$ per time step (one-factor); ensure positivity if normal models; use Newton-Raphson for implicit solutions.

\textcolor{blue}{\textbf{Finite Difference.}} More flexible than trees (accommodates wide variety of models) but slower convergence (requires discretization of entire state-space).

\textcolor{blue}{\textbf{1-D Explicit Scheme.}} One-factor diffusion under EMM: $dr_t=\mu(t,r_t)dt+\sigma(t,r_t)dW_t$. PDE for security value: $\frac{\partial g(t,r_t)}{\partial t}+\frac{\partial g(t,r_t)}{\partial r_t}\mu(t,r_t)+\frac{1}{2}\frac{\partial^2 g(t,r_t)}{\partial r_t^2}\sigma(t,r_t)^2-r_t g(t,r_t)=0$. Grid: space step $\Delta r$, time step $\Delta t$. Nodes indexed by $i$ (space, bottom to top), $t$ (time, current to maturity $T$). Discretized Greeks: $\Delta_i^t$, $\Gamma_i^t$, $\theta_i^t$ (delta, gamma, theta at node $i$, time $t$). Drift $\mu(t,r_t)$, vol $\sigma(t,r_t)$ discretized as $\mu_i^t$, $\sigma_i^t$. Discretized PDE: $\theta_i^t+\Delta_i^t\mu_i^t+\frac{1}{2}\Gamma_i^t(\sigma_i^t)^2-i\Delta r g=0$. Central diff. approx: $\Delta_i^t=\frac{g_{i+1}^t-g_{i-1}^t}{2\Delta r}$, $\Gamma_i^t=\frac{g_{i+1}^t-2g_i^t+g_{i-1}^t}{(\Delta r)^2}$, $\theta_i^t=\frac{g_i^t-g_i^{t-1}}{\Delta t}$. Backward recursion: $g_i^{t-1}=A_i^t g_{i-1}^{t-1}+B_i^t g_i^{t-1}+C_i^t g_{i+1}^{t-1}$ w/ $v_1=\frac{\Delta t}{(\Delta r)^2}$, $v_2=\frac{\Delta t}{\Delta r}$, coefs: $A_i^t=\frac{1}{2}[(\sigma_i^t)^2 v_1-\mu_i^t v_2]$, $B_i^t=1-(\sigma_i^t)^2 v_1-i\Delta r\Delta t$, $C_i^t=\frac{1}{2}[\mu_i^t v_2+(\sigma_i^t)^2 v_1]$. Properties: Sum $A_i^t+B_i^t+C_i^t=1-i\Delta r\Delta t$ captures 1-period discount; embeds martingale property. Weights $A_i^t,B_i^t,C_i^t$ not probabilities, may lie outside $[0,1]$, potentially causing instability. Stability requires small $v_1$, $v_2$: discretize time sufficiently relative to space (horizontally stretched grid). Choice of $r_{\max}$: balance truncation bias (need $r_{\max}$ high) vs computational efficiency (higher $r_{\max}$ requires finer discretization). Boundary conditions: ZC bonds: $g=1$ when $r=0$; $g=0$ when $r=r_{\max}$. Deep ITM: linear approx $g_{i_{\max}}^t=2g_{i_{\max}-1}^t-g_{i_{\max}-2}^t$. Deep OTM: derivatives worthless. American opts: compute max of exercise \& continuation values recursively at each node; use exercise value as boundary for deep ITM regions.

\textcolor{blue}{\textbf{2-D Explicit Scheme.}} Two-factor diffusion under EMM: $dx_s=\mu(t,x_s,y_t)dt+\sigma(t,x_s,y_t)dW_s^x$, $dy_t=\mu(t,x_s,y_t)dt+\sigma(t,x_s,y_t)dW_s^y$ w/ $dW_s^x dW_s^y=\rho dt$ (corr. $\rho$). Contingent claim value $g(t,x_s,y_t)$ satisfies: $\frac{\partial g}{\partial t}+\frac{\partial g}{\partial x_s}\mu(t,x_s,y_t)+\frac{1}{2}\frac{\partial^2 g}{\partial x_s^2}\sigma(t,x_s,y_t)^2+\frac{\partial g}{\partial y_t}\mu(t,x_s,y_t)+\frac{1}{2}\frac{\partial^2 g}{\partial y_t^2}\sigma(t,x_s,y_t)^2+\rho\sigma_x(t,x_s,y_t)\sigma_y(t,x_s,y_t)\frac{\partial^2 g}{\partial x_s\partial y_t}-r_t g(t,x_s,y_t)=0$. 3-D grid: space steps $\Delta x$, $\Delta y$; time step $\Delta t$. Notation: $g_{i,j}^t$=value at node $i$ for $x$ (left to right), node $j$ for $y$ (bottom to top), time $t$. Discretization: Apply central diffs for partial derivs \& backward recursion as 1-D case. Cross-partial deriv $\frac{\partial^2 g}{\partial x_s\partial y_t}$ discretized using 4-point stencil. Solution proceeds backward through time on 2-D grid at each time step.

\textcolor{blue}{\textbf{Monte Carlo.}} Generating normal: \textcolor{blue}{Box-Muller}: given $u_1,u_2$ uniform $[0,1]$: $\varepsilon_1=\sqrt{-2\ln(u_1)}\cos(2\pi u_2)$, $\varepsilon_2=\sqrt{-2\ln(u_1)}\sin(2\pi u_2)$. Euler discretization: $r_{t+\Delta t}=r_t+\kappa(\theta-r_t)\Delta t+\sigma\sqrt{\Delta t}\varepsilon_t$. Full truncation: replace $r_t$ w/ $\max(r_t,0)$ in drift \& diffusion to preserve non-negativity. Milstein: higher accuracy: $r_{t+\Delta t}=r_t+\kappa(\theta-r_t)\Delta t+\sigma\sqrt{\Delta t}\varepsilon_t+\frac{\sigma}{4}\Delta t(\varepsilon_t^2-1)$. Path-dependent valuation: For each path $j$, compute discount: $D^{(j)}=\exp(-\sum_{n=0}^{N-1}r_n^{(j)}\Delta t)$. Price security: $\text{Price}=\mathbb{E}^Q[D^{(j)}\cdot X(T)]=\frac{1}{M}\sum_{j=1}^M D^{(j)}\cdot X^{(j)}(T)$ where $X(T)$=terminal payoff, $M$=\# paths. Correlated BMs: Generate indep. $(\varepsilon_1,\varepsilon_2)$, then $\Delta W_1=\sqrt{\Delta t}\varepsilon_1$, $\Delta W_2=\sqrt{\Delta t}(\rho\varepsilon_1+\sqrt{1-\rho^2}\varepsilon_2)$.

\textcolor{blue}{\textbf{Path-Dependent Payoffs.}} Barrier caps/floors: Activated ("in") or deactivated ("out") first time ref. rate reaches critical barrier from below ("up") or above ("down"). Four types: up-in cap, down-in floor, down-out cap, up-out floor. Payoff: $(i_{u-1}^{(u)}-K)^+\cdot\mathbb{1}(\max_{s\in[u-1,u]}i_s^{(u)}>H)$. Barrier condition applies to whole cap/floor: once caplet activates, all subsequent activate. Bounded caps/floors: Total accumulated payout can't exceed $M$. At each settlement, accumulated value computed \& test applied to determine if below limit: $(i_{u-1}^{(u)}-K)^+\cdot\mathbb{1}(\sum_{j=1}^{u-1}(i_{j-1}^{(j)}-K)^+<M)$. Option deactivated first time test fails. Incremental fixed swaps: Fixed leg as weighted avg of fixed \& variable rates. Payer swap (notional $M$, fixed $K$, freq $\tau$): $-M\tau(\kappa+(1-x)u_t^{(u)})+M\tau d_t^{(u)}$. Fraction $x$ determined by ref. rate tier: $x=100\%$ if $i_t<i_1$; $x=60\%$ if $i_2\leq i_t\leq i_1$; $x=0\%$ if $i_t>i_2$. At each reset, simulated rate determines $x$ \& payoff. Ratchet caps/floors: Strike adjusts to level of ref. rate on preceding reset date. Payoff at settlement: $(i_{u-1}^{(u)}-i_{u-2}^{(u-1)})^+$. Simulation adjusted to compute diff between current \& previous reset rates at each settlement. Rolling caps/floors: Nominal not constant. When caplet (floorlet) not exercised (not ITM), its notional added to next. At each reset, apply test \& adjust payoff.

\textcolor{red}{\textbf{Derivatives Basics.}}
\textbf{Itô's product:} $dW(t)dW(t)=dt$, $dt\,dt=0$, $dW(t)dt=0$, $dW_1(t)dW_2(t)=\rho dt$.
\textbf{Itô's Lemma:} $df(t)=\frac{\partial f}{\partial t}dt+\frac{\partial f}{\partial x}dX(t)+\frac{1}{2}\frac{\partial^2 f}{\partial x^2}[dX(t)]^2$.
\textcolor{blue}{\textbf{Black-Scholes-Merton}} (Eur. opts): $c(t,T,K)=\hat{S}(t)\Phi(d_1)-Ke^{-r(T-t)}\Phi(d_2)$, $p(t,T,K)=Ke^{-r(T-t)}\Phi(-d_2)-\hat{S}(t)\Phi(-d_1)$ where $d_1=\frac{\ln(\hat{S}(t)/K)+(r+\frac{1}{2}\sigma^2)(T-t)}{\sigma\sqrt{T-t}}$, $d_2=d_1-\sigma\sqrt{T-t}$. Discrete dividends: $\hat{S}(t)=S(t)-\sum_{i=1}^n d(t_i)e^{-r(t_i-t)}$. Continuous yield: $\hat{S}(t)=S(t)e^{-y(T-t)}$.
\textbf{Pointers:} Normal PDF: $f(x)=\frac{1}{\sigma\sqrt{2\pi}}\exp(-\frac{(x-\mu)^2}{2\sigma^2})$. $\text{Var}(X)=\mathbb{E}[X^2]-(\mathbb{E}[X])^2$. If $x\sim\mathcal{N}(\mu,\sigma^2)$ then $x=\mu+\sigma\epsilon$, $\epsilon\sim\mathcal{N}(0,1)$, $\mathbb{E}[e^x]=e^{\mu+\frac{1}{2}\sigma^2}$.
